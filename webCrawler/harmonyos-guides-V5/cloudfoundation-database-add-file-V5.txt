引入对象类型文件
将导出的json格式文件命名为schema.json，拷贝到DevEco Studio项目的“entry/src/main/resources/rawfile”目录下。按照AGC控制台上创建的对象类型“BookInfo”在代码工程中创建BookInfo.ets文件，文件内容参考以下代码。import { cloudDatabase } from '@kit.CloudFoundationKit';\n\nclass BookInfo extends cloudDatabase.DatabaseObject{\n  public naturalbase_ClassName(): string {\n    return \"BookInfo\";\n  }\n  public id: number | undefined;\n  public bookName: string | undefined;\n  public author: string | undefined;\n  public price: number | undefined;\n  public publisher: string | undefined;\n  public publishTime: Date | undefined;\n}\n\nexport { BookInfo };
