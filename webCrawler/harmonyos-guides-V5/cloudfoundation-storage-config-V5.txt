设置云存储配置项
 在“entry/src/main/module.json5”文件中添加网络权限。\"requestPermissions\": [\n  {\n    \"name\": \"ohos.permission.INTERNET\"\n  }\n] 使用云存储服务，需要获取用户凭据。当前支持通过AGC认证服务SDK、或者华为账号服务Access Token接口两种方式获取，具体请参见AuthProvider。 （可选）配置云存储接口使用的网络类型，默认request.agent.Network.ANY。import { cloudCommon } from '@kit.CloudFoundationKit';\nimport { request } from '@kit.BasicServicesKit';\n\n// 设置云存储只使用wifi网络\ncloudCommon.init({\n  storageOptions: {\n    network: request.agent.Network.WIFI\n  }  \n}) （可选）配置云存储上传下载任务的工作模式，默认request.agent.Mode.BACKGROUND。import { cloudCommon } from '@kit.CloudFoundationKit';\nimport { request } from '@kit.BasicServicesKit';\n\n// 设置云存储上传下载任务为前台模式\ncloudCommon.init({\n  storageOptions: {\n    mode: request.agent.Mode.FOREGROUND\n  }\n})
